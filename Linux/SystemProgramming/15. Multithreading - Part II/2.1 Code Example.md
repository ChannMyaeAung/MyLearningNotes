# Code Example

- How do we recognize a critical section?
  - If we have the potential for parallelism and the code path is working on some shared resource (global or static data), then we have a critical section, which means we will protect it via locking.

```c
#define _GNU_SOURCE
#include <stdlib.h>
#include <stdio.h>
#include <assert.h>
#include <string.h>
#include <strings.h>
#include <sys/types.h>
#include <sys/time.h>
#include <unistd.h>
#include <sys/resource.h>
#include <sys/mman.h>
#include <alloca.h>
#include <malloc.h>
#include <fcntl.h>
#include <sys/capability.h>
#include <sys/wait.h>
#include <signal.h>
#include <errno.h>
#include <ctype.h>
#include <time.h>
#include <pthread.h>
#include "hello.h"

#define NUM_THREADS 10
#define NUM_ITERATIONS 100000

// Shared counter
int counter = 0;
// Mutex to protect the counter
pthread_mutex_t lock;

void *thread_func(void *arg)
{
    for (int i = 0; i < NUM_ITERATIONS; i++)
    {
        // Acquire the lock before entering the critical section
        pthread_mutex_lock(&lock);
        counter++; // Critical section: update shared resource
        // Release the lock after the update ensuring only one thread can modify counter at a time
        pthread_mutex_unlock(&lock);
    }
    return NULL;
}

int main(void)
{
    pthread_t threads[NUM_THREADS];

    // Initialize the mutex
    if (pthread_mutex_init(&lock, NULL) != 0)
    {
        perror("pthread_mutex_init");
        exit(EXIT_FAILURE);
    }

    // Create threads
    for (int i = 0; i < NUM_THREADS; i++)
    {
        if (pthread_create(&threads[i], NULL, thread_func, NULL) != 0)
        {
            perror("pthread_create");
            exit(EXIT_FAILURE);
        }
    }

    // Wait for all threads to finish
    for (int i = 0; i < NUM_THREADS; i++)
    {
        pthread_join(threads[i], NULL);
    }

    // Expected final counter value: NUM_THREADS * NUM_ITERATIONS
    printf("Final counter value: %d\n", counter);

    // Clean up the mutex
    pthread_mutex_destroy(&lock);

    return 0;
}
```
